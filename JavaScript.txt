
@section Scripts {

    <script type="text/javascript">

        // przekazanie wartości pobranej z selecta do tagu a, który ten z kolei przekazuje ją akcji GET Index


        let mainContainer = document.getElementById('containerIlinks');
        let selectList = document.getElementById('selectList');

        // Funkcja do stworzenia przycisku
        function createButton(container, text) {
            let aElement = document.createElement('a');
            aElement.style.padding = '5px';
            aElement.style.border = '1px solid rgb(220,220,220)';
            aElement.innerText = text;
            aElement.setAttribute('asp-route-SortowanieOption', selectList.value); // Przypisuje wartość z select

            // Listener dla kliknięcia
            aElement.addEventListener('click', function () {
                let sortowanieOptionText = this.getAttribute('asp-route-SortowanieOption');
                alert(sortowanieOptionText); // Wyświetla wartość atrybutu
            });

            container.appendChild(aElement);
        }
        createButton(mainContainer, 'ttttttttt');


        // Funkcja do tworzenia wielu przycisków
        function createXbuttons(container, iloscButtonow) {
            for (let i = 0; i < iloscButtonow; i++) {
                createButton(container, `Button ${i + 1}`);
            }
        }

        // Wywołanie do stworzenia przycisków
        createXbuttons(mainContainer, 10);

        // Listener do zmiany w select
        selectList.addEventListener('change', function () {
            let buttons = mainContainer.querySelectorAll('a');
            buttons.forEach(button => {
                button.setAttribute('asp-route-SortowanieOption', this.value); // Uaktualnia atrybut dla wszystkich przycisków
            });
        });

    </script>
}
