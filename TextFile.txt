
@section Scripts {

    @*Skrypt pozwalający zapamiętać wybrane zdjęcia po przeładowaniu strony*@ 
    
    <script>
        var filesInput = document.getElementById('files');
        var imagePhoto = document.getElementById('imagePhoto');

        // Funkcja do zapisywania przesłanych plików w sessionStorage
        function saveFiles() {
            var files = filesInput.files; // pobranie danych z inputa

            if (files.length > 0) {
                // Tworzenie URL dla pierwszego pliku
                var file = files[0];
                var fileURL = URL.createObjectURL(file);

                // Zapisanie URL w sessionStorage
                sessionStorage.setItem('imageURL', fileURL);

                // Ustawienie podglądu zdjęcia
                imagePhoto.src = fileURL;
                imagePhoto.style.width = '150px';
                imagePhoto.style.height = '120px';
            }
        }

        window.onload = function () {
            // Sprawdzenie, czy w sessionStorage znajduje się zapisany URL
            var savedImageURL = sessionStorage.getItem('imageURL');
            if (savedImageURL) {
                // Ustawienie zdjęcia na podstawie URL z sessionStorage
                imagePhoto.src = savedImageURL;
                imagePhoto.style.width = '150px';
                imagePhoto.style.height = '120px';
            }
        };
    </script>
}
